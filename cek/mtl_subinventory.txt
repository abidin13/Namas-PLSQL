SELECT msub.secondary_inventory_name, msub.locator_type, msub.description FROM mtl_secondary_inventories msub WHERE msub.organization_id = NVL (:po_startup_values.org_id, msub.organization_id) AND NVL (msub.disable_date, TRUNC (SYSDATE + 1)) > TRUNC (SYSDATE) AND (:rcv_transaction.item_id IS NULL OR (:rcv_transaction.item_id IS NOT NULL AND EXISTS (SELECT NULL FROM mtl_system_items msi WHERE msi.organization_id = NVL (:po_startup_values.org_id,msi.organization_id) AND msi.inventory_item_id =                                  :rcv_transaction.item_id AND (msi.restrict_subinventories_code = 2 OR (msi.restrict_subinventories_code = 1 AND EXISTS (SELECT NULL FROM mtl_item_sub_inventories mis WHERE mis.organization_id = NVL (:po_startup_values.org_id,                                                              mis.                                                                organization_id)                                                         AND mis.                                                              inventory_item_id =                                                                msi.                                                                 inventory_item_id                                                         AND mis.                                                              secondary_inventory =                                                                msub.                                                                 secondary_inventory_name))))))         AND (msub.secondary_inventory_name LIKE '%STI%' OR msub.secondary_inventory_name like '%SPR%' OR msub.secondary_inventory_name like '%PRD%' OR msub.secondary_inventory_name LIKE '%RMT%')         AND msub.attribute1 = ynpinv_get_resp_fnc ORDER BY msub.secondary_inventory_name
