SELECT(SELECT NAME FROM HR_ALL_ORGANIZATION_UNITS WHERE ORGANIZATION_ID = A.ORG_ID) OU,
        MCBV.SEGMENT1 PRODUK,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/01/01', '2015/01/31') FROM dual) AS JAN,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/02/01', '2015/02/28') FROM dual) AS FEB,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/03/01', '2015/03/31') FROM dual) AS MAR,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/04/01', '2015/04/30') FROM dual) AS APR,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/05/01', '2015/05/31') FROM dual) AS MAY,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/06/01', '2015/06/30') FROM dual) AS JUN,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/07/01', '2015/07/31') FROM dual) AS JUL,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/08/01', '2015/08/30') FROM dual) AS AUG,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/09/01', '2015/09/30') FROM dual) AS SEP,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/10/01', '2015/10/31') FROM dual) AS OKT,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/11/01', '2015/11/30') FROM dual) AS NOV,
        (SELECT YNP_GET_ORDERBULANAN(A.ORG_ID, MCBV.SEGMENT1, '2015/12/01', '2015/12/31') FROM dual) AS DES       
FROM OE_ORDER_HEADERS_ALL A, OE_ORDER_LINES_ALL B, MTL_ITEM_CATEGORIES MICV, MTL_CATEGORIES_B_KFV MCBV
WHERE A.HEADER_ID = B.HEADER_ID
AND B.INVENTORY_ITEM_ID = MICV.INVENTORY_ITEM_ID
AND MICV.CATEGORY_ID = MCBV.CATEGORY_ID
AND MICV.ORGANIZATION_ID = 89
AND MICV.CATEGORY_SET_ID = '1100000050'
AND B.SHIPPING_INTERFACED_FLAG = 'Y'
--AND A.ORG_ID = :ORG_ID

GROUP BY A.ORG_ID, MCBV.SEGMENT1
ORDER BY OU asc
